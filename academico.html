#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "PROCESS_INTERNAL";
const char* password = "CRYPT_PASSWORD";

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  // Inicialização do sensor DS18B20
}

void loop() {
  float tempC = readTemperature(); // Função de leitura

  if(WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin("http://sql.process.local/data");
    http.addHeader("Content-Type", "application/json");
    
    String json = "{\"sensor\": \"FERM_01\", \"val\": " + String(tempC) + "}";
    int response = http.POST(json);
    http.end();
  }
  delay(5000); 
}
